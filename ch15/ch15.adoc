=== Exercise: Optional Monoid

[source, haskell]
instance Monoid a => Monoid (Optional a) where
  mempty = mempty
  mappend Nada Nada = Nada
  mappend Nada (Only x) = Only x
  mappend (Only x) Nada = Only x
  mappend (Only x) (Only y) = Only (x `mappend` y)
